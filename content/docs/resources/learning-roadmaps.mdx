### Code Patterns That Save Hours

#### The "Never Loading" Pattern

<Callout type="warn">
  Stop writing loading states that flash for 200ms. It looks janky!
</Callout>

```tsx
// ❌ BAD: Flashing loading states
function UserProfile({ userId }) {
  const { data, isLoading } = useQuery(["user", userId], getUser);

  if (isLoading) return <Spinner />; // Flashes for 200ms 😬

  return <Profile user={data} />;
}

// ✅ GOOD: Use stale-while-revalidate + minimum loading time
function UserProfile({ userId }) {
  const { data, isLoading, isFetching } = useQuery(["user", userId], getUser, {
    staleTime: 5 * 60 * 1000, // Data is "fresh" for 5 mins
    keepPreviousData: true, // Show old data while fetching new
  });

  return (
    <div className="relative">
      {isFetching && <SubtleSpinner />} {/* Subtle indicator */}
      <Profile user={data} />
    </div>
  );
}
```

#### The Compound Component Pattern

```tsx
// ❌ BAD: Prop drilling nightmare
<Menu
  items={items}
  selectedItem={selected}
  onSelect={setSelected}
  renderItem={(item) => <MenuItem {...item} />}
  renderTrigger={(props) => <Button {...props} />}
/>

// ✅ GOOD: Compound components are composable
<Menu>
  <Menu.Trigger>
    <Button>Open Menu</Button>
  </Menu.Trigger>
  <Menu.Content>
    <Menu.Item onSelect={() => edit()}>Edit</Menu.Item>
    <Menu.Item onSelect={() => duplicate()}>Duplicate</Menu.Item>
    <Menu.Separator />
    <Menu.Item variant="destructive" onSelect={() => remove()}>
      Delete
    </Menu.Item>
  </Menu.Content>
</Menu>
```

### TypeScript Wizardry

#### Utility Types You Should Use More

```typescript
// 1. Partial<T> - All properties optional
type User = { id: string; name: string; email: string };
type PartialUser = Partial<User>;

// Great for update functions
function updateUser(id: string, updates: Partial<User>) {
  // Can pass just { name: "New Name" }
}

// 2. Pick<T, K> - Select specific properties
type UserPreview = Pick<User, "id" | "name">;

// 3. Omit<T, K> - Remove specific properties
type CreateUser = Omit<User, "id">;

// 4. Record<K, T> - Dynamic object types
type StatusColors = Record<"success" | "error" | "warning", string>;

// 5. ReturnType<T> - Extract return type of function
type ApiResponse = ReturnType<typeof fetchUser>;

// 6. Parameters<T> - Extract parameter types
type FetchUserParams = Parameters<typeof fetchUser>;
```

#### Discriminated Unions for Type-Safe State

```typescript
// ❌ BAD: Optional properties everywhere
type RequestState = {
  data?: User;
  error?: Error;
  isLoading?: boolean;
};
// Problem: Can have data AND error at same time!

// ✅ GOOD: Discriminated union
type RequestState =
  | { status: "idle" }
  | { status: "loading" }
  | { status: "success"; data: User }
  | { status: "error"; error: Error };

// Now TypeScript knows exactly what's available
function handleState(state: RequestState) {
  switch (state.status) {
    case "loading":
      return <Spinner />;
    case "success":
      return <UserCard user={state.data} />; // TS knows data exists!
    case "error":
      return <Error message={state.error.message} />; // TS knows error exists!
    default:
      return null;
  }
}
```

#### The `as const` Trick

```typescript
// Without `as const`
const ROUTES = {
  home: "/",
  dashboard: "/dashboard",
  profile: "/profile",
}; // Type: { home: string; dashboard: string; ... }

// With `as const`
const ROUTES = {
  home: "/",
  dashboard: "/dashboard",
  profile: "/profile",
} as const; // Type: { readonly home: "/"; readonly dashboard: "/dashboard"; ... }

// Now you can extract literal types!
type Route = (typeof ROUTES)[keyof typeof ROUTES];
// "/" | "/dashboard" | "/profile"
```

## Performance Tips

### React Performance Checklist

<Steps>
<Step>
**Identify Re-renders with DevTools**

```tsx
// Add this in development to spot unnecessary re-renders
import { useEffect, useRef } from "react";

function useWhyDidYouUpdate(name: string, props: Record<string, any>) {
  const previousProps = useRef<Record<string, any>>({});

  useEffect(() => {
    if (previousProps.current) {
      const changedProps = Object.entries(props).filter(
        ([key, value]) => previousProps.current[key] !== value
      );
      if (changedProps.length > 0) {
        console.log(`[${name}] Changed props:`, changedProps);
      }
    }
    previousProps.current = props;
  });
}
```

</Step>

<Step>
**Memoize Expensive Calculations**

```tsx
// ❌ Recalculates on every render
const sortedItems = items.sort((a, b) => a.name.localeCompare(b.name));

// ✅ Only recalculates when items change
const sortedItems = useMemo(
  () => items.sort((a, b) => a.name.localeCompare(b.name)),
  [items]
);
```

</Step>

<Step>
**Use React.memo for Pure Components**

```tsx
// For components that render the same output for same props
const UserCard = React.memo(function UserCard({ user }) {
  return (
    <div>
      <Avatar src={user.avatar} />
      <span>{user.name}</span>
    </div>
  );
});

// Custom comparison for complex props
const UserCard = React.memo(
  function UserCard({ user }) {
    /* ... */
  },
  (prevProps, nextProps) => prevProps.user.id === nextProps.user.id
);
```

</Step>

<Step>
**Virtualize Long Lists**

```tsx
import { useVirtualizer } from "@tanstack/react-virtual";

function VirtualList({ items }) {
  const parentRef = useRef(null);

  const virtualizer = useVirtualizer({
    count: items.length,
    getScrollElement: () => parentRef.current,
    estimateSize: () => 50,
  });

  return (
    <div ref={parentRef} className="h-[400px] overflow-auto">
      <div style={{ height: `${virtualizer.getTotalSize()}px` }}>
        {virtualizer.getVirtualItems().map((virtualRow) => (
          <div
            key={virtualRow.key}
            style={{
              height: `${virtualRow.size}px`,
              transform: `translateY(${virtualRow.start}px)`,
            }}
          >
            {items[virtualRow.index].name}
          </div>
        ))}
      </div>
    </div>
  );
}
```

</Step>
</Steps>

## Git Power Moves

```bash
# 1. Undo last commit but keep changes
git reset --soft HEAD~1

# 2. Amend last commit message
git commit --amend -m "New message"

# 3. Interactive rebase (squash commits)
git rebase -i HEAD~3

# 4. Stash with message
git stash push -m "WIP: auth feature"

# 5. Cherry-pick specific commit
git cherry-pick abc123

# 6. See who wrote each line
git blame filename.ts

# 7. Search all commits for a string
git log -S "searchTerm" --oneline

# 8. Pretty log with graph
git log --oneline --graph --all

# 9. Undo a pushed commit (safe)
git revert abc123

# 10. Clean up local branches
git branch --merged | grep -v main | xargs git branch -d
```
