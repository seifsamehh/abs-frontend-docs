<Step>
### ğŸ“ Plan Your Component

<Step>
### ğŸ“ Create the Component File

Create a new file following our naming convention:

```bash
# For a UserCard component
src/components/user-card.tsx
```

Start with this template:

```tsx
import * as React from "react";
import { cn } from "@/lib/utils";

interface UserCardProps extends React.HTMLAttributes<HTMLDivElement> {
  /** The user's display name */
  name: string;
  /** The user's email address */
  email: string;
  /** Optional avatar URL */
  avatarUrl?: string;
}

/**
 * UserCard displays user information in a card format.
 *
 * @example
 * <UserCard name="John Doe" email="john@example.com" />
 */
export function UserCard({
  name,
  email,
  avatarUrl,
  className,
  ...props
}: UserCardProps) {
  return (
    <div
      className={cn(
        "rounded-lg border bg-card p-4 text-card-foreground shadow-sm",
        className
      )}
      {...props}
    >
      <div className="flex items-center gap-4">
        {avatarUrl && (
          <img
            src={avatarUrl}
            alt={`${name}'s avatar`}
            className="h-12 w-12 rounded-full object-cover"
          />
        )}
        <div>
          <h3 className="font-semibold">{name}</h3>
          <p className="text-sm text-muted-foreground">{email}</p>
        </div>
      </div>
    </div>
  );
}
```

</Step>

<Step>
### ğŸ­ Add Component Variants

Use `class-variance-authority` for variant management:

```tsx
import { cva, type VariantProps } from "class-variance-authority";

const userCardVariants = cva(
  "rounded-lg border p-4 text-card-foreground shadow-sm",
  {
    variants: {
      variant: {
        default: "bg-card",
        highlighted: "bg-primary/10 border-primary",
        muted: "bg-muted",
      },
      size: {
        sm: "p-2",
        md: "p-4",
        lg: "p-6",
      },
    },
    defaultVariants: {
      variant: "default",
      size: "md",
    },
  }
);

interface UserCardProps
  extends React.HTMLAttributes<HTMLDivElement>,
    VariantProps<typeof userCardVariants> {
  name: string;
  email: string;
  avatarUrl?: string;
}

export function UserCard({
  name,
  email,
  avatarUrl,
  variant,
  size,
  className,
  ...props
}: UserCardProps) {
  return (
    <div
      className={cn(userCardVariants({ variant, size }), className)}
      {...props}
    >
      {/* ... component content */}
    </div>
  );
}
```

</Step>

<Step>
### â™¿ Ensure Accessibility

Always include proper accessibility attributes:

```tsx
export function UserCard({
  name,
  email,
  avatarUrl,
  className,
  ...props
}: UserCardProps) {
  return (
    <article
      className={cn("rounded-lg border bg-card p-4 shadow-sm", className)}
      aria-label={`User card for ${name}`}
      {...props}
    >
      <div className="flex items-center gap-4">
        {avatarUrl && (
          <img
            src={avatarUrl}
            alt="" // Decorative, name is already announced
            aria-hidden="true"
            className="h-12 w-12 rounded-full object-cover"
          />
        )}
        <div>
          <h3 className="font-semibold">{name}</h3>
          <p className="text-sm text-muted-foreground">
            <span className="sr-only">Email: </span>
            {email}
          </p>
        </div>
      </div>
    </article>
  );
}
```

</Step>

<Step>
### ğŸ§ª Write Tests

Create a test file alongside your component:

```tsx
// src/components/user-card.test.tsx
import { render, screen } from "@testing-library/react";
import { UserCard } from "./user-card";

describe("UserCard", () => {
  it("renders user name and email", () => {
    render(<UserCard name="John Doe" email="john@example.com" />);

    expect(screen.getByText("John Doe")).toBeInTheDocument();
    expect(screen.getByText("john@example.com")).toBeInTheDocument();
  });

  it("renders avatar when provided", () => {
    render(
      <UserCard
        name="John Doe"
        email="john@example.com"
        avatarUrl="/avatar.jpg"
      />
    );

    expect(screen.getByRole("img")).toHaveAttribute("src", "/avatar.jpg");
  });

  it("applies custom className", () => {
    const { container } = render(
      <UserCard name="John" email="john@test.com" className="custom-class" />
    );

    expect(container.firstChild).toHaveClass("custom-class");
  });
});
```

</Step>

## ğŸ“¦ Component Categories

### ğŸ“ Layout Components

## â­ Best Practices

<Callout type="warn">
  Always follow these practices when building components.
</Callout>

### Do's âœ…

- Use TypeScript for all props with JSDoc comments
- Forward refs for DOM element components
- Use the `cn()` utility for className merging
- Include proper ARIA attributes
- Make components keyboard navigable
- Write unit tests for all components

### Don'ts âŒ

- Don't hardcode colors - use CSS variables
- Don't ignore accessibility requirements
- Don't create overly complex components
- Don't skip TypeScript types
- Don't forget to handle loading/error states
