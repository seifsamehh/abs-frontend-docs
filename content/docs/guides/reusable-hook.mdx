## Problem

You need a reusable, type-safe hook for fetching data that handles loading, error, and caching states consistently across components.

## Solution

<Tabs items={["Hook", "Usage"]}>
<Tab value="Hook">
```ts title="hooks/use-projects.ts"
import { useQuery, queryOptions } from "@tanstack/react-query";
import { z } from "zod";

const projectSchema = z.object({
  id: z.string(),
  name: z.string(),
  status: z.enum(["active", "archived"]),
});

type Project = z.infer<typeof projectSchema>;

// Export query options for prefetching / cache invalidation
export const projectsQueryOptions = (teamId: string) =>
  queryOptions({
    queryKey: ["projects", teamId],
    queryFn: async (): Promise<Project[]> => {
      const res = await fetch(`/api/teams/${teamId}/projects`);
      if (!res.ok) throw new Error("Failed to fetch projects");
      const data = await res.json();
      return z.array(projectSchema).parse(data);
    },
    staleTime: 5 * 60 * 1000, // 5 minutes
  });

export function useProjects(teamId: string) {
  return useQuery(projectsQueryOptions(teamId));
}
```
</Tab>
<Tab value="Usage">
```tsx title="components/project-list.tsx"
import { useProjects } from "@/hooks/use-projects";

export function ProjectList({ teamId }: { teamId: string }) {
  const { data: projects, isLoading, error } = useProjects(teamId);

  if (isLoading) return <div>Loading...</div>;
  if (error) return <div>Error: {error.message}</div>;

  return (
    <ul>
      {projects?.map((p) => (
        <li key={p.id}>{p.name}</li>
      ))}
    </ul>
  );
}
```
</Tab>

## Tips

<Callout type="info">
Export `queryOptions` separately so you can reuse the same query key for prefetching in loaders and for cache invalidation after mutations.
</Callout>

- Always validate API responses with Zod -- the server contract can change without warning.
- Use `staleTime` to avoid unnecessary refetches; set it based on how frequently your data changes.
- Name hooks `use<Resource>` and keep one hook per file for discoverability.
