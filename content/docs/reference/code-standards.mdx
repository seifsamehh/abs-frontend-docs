### Component Template

```tsx
import * as React from "react";
import { cn } from "@/lib/utils";

interface UserCardProps extends React.HTMLAttributes<HTMLDivElement> {
  /** User data to display */
  user: User;
  /** Callback when edit button is clicked */
  onEdit?: (user: User) => void;
  /** Callback when delete button is clicked */
  onDelete?: (user: User) => void;
}

/**
 * UserCard displays user information with optional action buttons.
 *
 * @example
 * ```tsx
 * <UserCard user={userData} onEdit={handleEdit} />
 * ```
 */
export function UserCard({
  user,
  onEdit,
  onDelete,
  className,
  ...props
}: UserCardProps) {
  return (
    <div
      className={cn(
        "rounded-lg border bg-card p-4 shadow-sm",
        className
      )}
      {...props}
    >
      <h3 className="font-semibold">{user.name}</h3>
      <p className="text-sm text-muted-foreground">{user.email}</p>
      {(onEdit || onDelete) && (
        <div className="mt-4 flex gap-2">
          {onEdit && (
            <button onClick={() => onEdit(user)}>Edit</button>
          )}
          {onDelete && (
            <button onClick={() => onDelete(user)}>Delete</button>
          )}
        </div>
      )}
    </div>
  );
}

## ğŸ£ Custom Hooks Standards

````tsx
import { useState, useEffect, useCallback } from "react";

interface UseUserDataOptions {
  /** Initial user ID to fetch */
  userId?: string;
  /** Auto-refetch interval in milliseconds */
  refetchInterval?: number;
}

interface UseUserDataReturn {
  /** User data */
  user: User | null;
  /** Loading state */
  isLoading: boolean;
  /** Error message if any */
  error: string | null;
  /** Function to manually refetch */
  refetch: () => void;
}

/**
 * Custom hook for fetching and managing user data.
 *
 * @param options - Configuration options
 * @returns Object containing user data, loading state, error, and refetch function
 *
 * @example
 * ```tsx
 * const { user, isLoading, error, refetch } = useUserData({ userId: "123" });
 * ```
 */
export function useUserData(
  options: UseUserDataOptions = {}
): UseUserDataReturn {
  const { userId, refetchInterval } = options;
  const [user, setUser] = useState<User | null>(null);
  const [isLoading, setIsLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const refetch = useCallback(async () => {
    if (!userId) return;

    setIsLoading(true);
    setError(null);

    try {
      const data = await fetchUser(userId);
      setUser(data);
    } catch (err) {
      setError(err instanceof Error ? err.message : "Failed to fetch user");
    } finally {
      setIsLoading(false);
    }
  }, [userId]);

  useEffect(() => {
    refetch();
  }, [refetch]);

  useEffect(() => {
    if (!refetchInterval) return;

    const interval = setInterval(refetch, refetchInterval);
    return () => clearInterval(interval);
  }, [refetch, refetchInterval]);

  return { user, isLoading, error, refetch };
}
````

## ğŸ”’ Security Standards

### Input Validation

```tsx
import { z } from "zod";

// Always validate user input
const userSchema = z.object({
  name: z.string().min(1).max(100),
  email: z.string().email(),
  age: z.number().int().positive().max(150),
});

// Validate before using
const validatedData = userSchema.parse(userInput);

### XSS Prevention

```tsx
// âœ… React automatically escapes
<p>{userInput}</p>

// âŒ Never use dangerouslySetInnerHTML with user input
<div dangerouslySetInnerHTML={{ __html: userInput }} />

// âœ… If you must render HTML, sanitize first
import DOMPurify from "dompurify";
<div dangerouslySetInnerHTML={{ __html: DOMPurify.sanitize(html) }} />
```

### Sensitive Data

```tsx

## â™¿ Accessibility Standards

### Required Practices

| Requirement        | Implementation                                       |
| ------------------ | ---------------------------------------------------- |
| **Labels**         | All form inputs must have associated labels          |
| **Alt text**       | All images must have alt attributes                  |
| **Focus**          | All interactive elements must be keyboard accessible |
| **ARIA**           | Use ARIA attributes for custom components            |
| **Color contrast** | Minimum 4.5:1 for normal text                        |
| **Focus visible**  | Visible focus indicators on all interactive elements |

```tsx
// âœ… Accessible button
<button
  onClick={handleClick}
  disabled={isLoading}
  aria-busy={isLoading}
  aria-label="Submit form"
>
  {isLoading ? "Loading..." : "Submit"}
</button>

// âœ… Accessible form field
<div>
  <label htmlFor="email">Email</label>
  <input
    id="email"
    type="email"
    aria-required="true"
    aria-invalid={!!error}
    aria-describedby={error ? "email-error" : undefined}
  />
  {error && <p id="email-error" role="alert">{error}</p>}
</div>
```
